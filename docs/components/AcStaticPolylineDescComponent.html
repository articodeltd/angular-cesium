src/components/ac-static-polyline-desc/ac-static-polyline-desc.component.ts

Description

This is a static implementation of an polyline.
usage:

   <ac-static-polyline
           geometryProps="{
               width: poly.geometry.width,
               positions: poly.geometry.positions
           }"
           instanceProps="{
             attributes: {
                 Cesium.ColorGeometryInstanceAttribute.fromColor(Cesium.Color.fromRandom())
             }
           }"
           primitiveProps="{
             appearance: new Cesium.PolylineColorAppearance()
   }"></ac-static-polyline>
